import historyModel from "../../models/historyModel.js";
import error from "../../helpers/errors.js";


/**
 * Finds all history entries in the database and returns them sorted by date (newest first) and by schedule hours (ascending).
 * @returns {Promise<Array<import('../../models/historyModel').History>>} A promise that resolves to an array of history entries.
 * @throws {error.FINDALL_EMPTY} If no history entries were found.
 */
async function getAllHistory() {
    const history = await historyModel
    .find()
    .sort({
        date: -1,
        'schedule.hours': 1
    })
    if (!history) {
        throw new error.FINDALL_EMPTY();
    }
    return history;
}

export const functions = {
    getAllHistory,
}
export default functions
